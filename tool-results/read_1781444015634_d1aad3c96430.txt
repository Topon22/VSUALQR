    70тЖТ    70тЖТconst fileToBase64 = (file: File): Promise<string> => {
    71тЖТ    71тЖТ  return new Promise((resolve, reject) => {
    72тЖТ    72тЖТ    const reader = new FileReader();
    73тЖТ    73тЖТ    reader.readAsDataURL(file);
    74тЖТ    74тЖТ    reader.onload = () => {

    76тЖТ    76тЖТ      const data = base64.split(',')[1];
    77тЖТ    77тЖТ      resolve(data);
    78тЖТ    78тЖТ    };
    79тЖТ    79тЖТ    reader.onerror = reject;
    80тЖТ    80тЖТ  });
    81тЖТ    81тЖТ};

    99тЖТ    99тЖТconst compressImage = (base64: string, maxSize = 800, quality = 0.8): Promise<string> => {
   100тЖТ   100тЖТ  return new Promise((resolve) => {
   101тЖТ   101тЖТ    const img = new window.Image();
   102тЖТ   102тЖТ    img.onload = () => {
   103тЖТ   103тЖТ      const canvas = document.createElement('canvas');
   104тЖТ   104тЖТ      let w = img.width;
   105тЖТ   105тЖТ      let h = img.height;
   106тЖТ   106тЖТ      if (w > maxSize || h > maxSize) {
   107тЖТ   107тЖТ        if (w > h) { h = Math.round((h / w) * maxSize); w = maxSize; }
   108тЖТ   108тЖТ        else { w = Math.round((w / h) * maxSize); h = maxSize; }
   109тЖТ   109тЖТ      }
   110тЖТ   110тЖТ      canvas.width = w;
   111тЖТ   111тЖТ      canvas.height = h;
   112тЖТ   112тЖТ      const ctx = canvas.getContext('2d');
   113тЖТ   113тЖТ      if (ctx) {
   114тЖТ   114тЖТ        ctx.drawImage(img, 0, 0, w, h);
   115тЖТ   115тЖТ        resolve(canvas.toDataURL('image/jpeg', quality).split(',')[1]);
   116тЖТ   116тЖТ      } else {
   117тЖТ   117тЖТ        resolve(base64);
   118тЖТ   118тЖТ      }
   119тЖТ   119тЖТ    };
   120тЖТ   120тЖТ    img.onerror = () => resolve(base64);
   121тЖТ   121тЖТ    img.src = `data:image/jpeg;base64,${base64}`;
   122тЖТ   122тЖТ  });
   123тЖТ   123тЖТ};

   252тЖТ   252тЖТfunction SplashScreen({ onComplete }: { onComplete: () => void }) {
   253тЖТ   253тЖТ  const [fading, setFading] = useState(false);
   254тЖТ   254тЖТ  const [phase, setPhase] = useState<'loading' | 'ready' | 'playing' | 'blocked'>('loading');
   255тЖТ   255тЖТ  const [videoProgress, setVideoProgress] = useState(0);
   256тЖТ   256тЖТ  const completedRef = useRef(false);
   257тЖТ   257тЖТ  const videoRef = useRef<HTMLVideoElement>(null);
   258тЖТ   258тЖТ  const device = useDeviceType();
   259тЖТ   259тЖТ
   260тЖТ   260тЖТ  const videoSrc = device === 'mobile' ? '/red-720p.mp4' : '/red.mp4';
   261тЖТ   261тЖТ  const videoDuration = 4; // seconds
   262тЖТ   262тЖТ
   263тЖТ   263тЖТ  const triggerComplete = useCallback(() => {
   264тЖТ   264тЖТ    if (!completedRef.current) {
   265тЖТ   265тЖТ      completedRef.current = true;
   266тЖТ   266тЖТ      setFading(true);
   267тЖТ   267тЖТ      setTimeout(onComplete, 500);
   268тЖТ   268тЖТ    }
   269тЖТ   269тЖТ  }, [onComplete]);
   270тЖТ   270тЖТ
   271тЖТ   271тЖТ  // Video loaded enough to play тАФ try autoplay
   272тЖТ   272тЖТ  const handleLoadedData = useCallback(() => {
   273тЖТ   273тЖТ    const vid = videoRef.current;
   274тЖТ   274тЖТ    if (!vid) return;
   275тЖТ   275тЖТ    vid.play().then(() => {
   276тЖТ   276тЖТ      setPhase('playing');
   277тЖТ   277тЖТ    }).catch(() => {
   278тЖТ   278тЖТ      // Autoplay blocked (iOS Safari, some Android)
   279тЖТ   279тЖТ      setPhase('blocked');
   280тЖТ   280тЖТ    });
   281тЖТ   281тЖТ  }, []);
   282тЖТ   282тЖТ
   283тЖТ   283тЖТ  // Progress tracking
   284тЖТ   284тЖТ  const handleTimeUpdate = useCallback(() => {
   285тЖТ   285тЖТ    const vid = videoRef.current;
   286тЖТ   286тЖТ    if (vid && vid.duration > 0) {
   287тЖТ   287тЖТ      setVideoProgress((vid.currentTime / vid.duration) * 100);
   288тЖТ   288тЖТ    }
   289тЖТ   289тЖТ  }, []);
   290тЖТ   290тЖТ
   291тЖТ   291тЖТ  // Video ended тАФ auto-dismiss
   292тЖТ   292тЖТ  const handleVideoEnd = useCallback(() => {
   293тЖТ   293тЖТ    triggerComplete();
   294тЖТ   294тЖТ  }, [triggerComplete]);
   295тЖТ   295тЖТ
   296тЖТ   296тЖТ  // Tap handler: play if blocked, dismiss if playing
   297тЖТ   297тЖТ  const handleTap = useCallback(() => {
   298тЖТ   298тЖТ    if (phase === 'blocked') {
   299тЖТ   299тЖТ      const vid = videoRef.current;
   300тЖТ   300тЖТ      if (vid) {
   301тЖТ   301тЖТ        vid.play().then(() => setPhase('playing')).catch(() => triggerComplete());
   302тЖТ   302тЖТ      }
   303тЖТ   303тЖТ      return;
   304тЖТ   304тЖТ    }
   305тЖТ   305тЖТ    triggerComplete();
   306тЖТ   306тЖТ  }, [phase, triggerComplete]);
   307тЖТ   307тЖТ
   308тЖТ   308тЖТ  // Auto-play attempt with loadeddata (more reliable than canplaythrough for short videos)
   309тЖТ   309тЖТ  useEffect(() => {
   310тЖТ   310тЖТ    const vid = videoRef.current;
   311тЖТ   311тЖТ    if (!vid) return;
   312тЖТ   312тЖТ
   313тЖТ   313тЖТ    // loadeddata fires when enough data is available to play
   314тЖТ   314тЖТ    const onLoaded = () => {
   315тЖТ   315тЖТ      setPhase('ready');
   316тЖТ   316тЖТ      vid.play().then(() => {
   317тЖТ   317тЖТ        setPhase('playing');
   318тЖТ   318тЖТ      }).catch(() => {
   319тЖТ   319тЖТ        setPhase('blocked');
   320тЖТ   320тЖТ      });
   321тЖТ   321тЖТ    };
   322тЖТ   322тЖТ    vid.addEventListener('loadeddata', onLoaded);
   323тЖТ   323тЖТ
   324тЖТ   324тЖТ    return () => vid.removeEventListener('loadeddata', onLoaded);
   325тЖТ   325тЖТ  }, [videoSrc]); // re-bind when src changes
   326тЖТ   326тЖТ
   327тЖТ   327тЖТ  // Safety timeout: dismiss after 6s
   328тЖТ   328тЖТ  useEffect(() => {
   329тЖТ   329тЖТ    const timer = setTimeout(() => {
   330тЖТ   330тЖТ      if (!completedRef.current) triggerComplete();
   331тЖТ   331тЖТ    }, 6000);
   332тЖТ   332тЖТ    return () => clearTimeout(timer);
   333тЖТ   333тЖТ  }, [triggerComplete]);
   334тЖТ   334тЖТ
   335тЖТ   335тЖТ  // Keyboard: Escape or Enter to skip
   336тЖТ   336тЖТ  useEffect(() => {
   337тЖТ   337тЖТ    const onKey = (e: KeyboardEvent) => {
   338тЖТ   338тЖТ      if (e.key === 'Escape' || e.key === 'Enter') {
   339тЖТ   339тЖТ        e.preventDefault();
   340тЖТ   340тЖТ        triggerComplete();
   341тЖТ   341тЖТ      }
   342тЖТ   342тЖТ    };
   343тЖТ   343тЖТ    window.addEventListener('keydown', onKey);
   344тЖТ   344тЖТ    return () => window.removeEventListener('keydown', onKey);
   345тЖТ   345тЖТ  }, [triggerComplete]);
   346тЖТ   346тЖТ
   347тЖТ   347тЖТ  return (
   348тЖТ   348тЖТ    <div
   349тЖТ   349тЖТ      className={`splash-container ${fading ? 'splash-exit' : ''}`}
   350тЖТ   350тЖТ      onClick={handleTap}
   351тЖТ   351тЖТ      style={{ touchAction: 'manipulation', cursor: 'pointer' }}
   352тЖТ   352тЖТ      role="dialog"
   353тЖТ   353тЖТ      aria-label="Loading VSUAL"
